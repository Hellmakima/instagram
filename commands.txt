// login with git GUI.

// always pull first.
git pull origin main

// in /frontend
npm install // only first time
npm run dev // to run
^C // to stop

// when updating
git add .
git commit -m "your message"
git push origin main // main is branch name

// new NextJS app
npx create-next-app@latest frontend

git branch // to check curr branch
git branch -a // list all branches
git branch devde // devde is new branch name
git checkout devde // change branch
// merge/ pull from github web interface
git branch -D devde // delete branch


// back
cloudflared tunnel --url http://localhost:5000
uvicorn main:app --reload --port 5000
